47      {

54          CompilerMessage compilerError = JavacCompiler.parseModernError( 0, error );

56          assertNotNull( compilerError );

58          assertFalse( compilerError.isError() );

62          assertEquals( 63, compilerError.getStartColumn() );
63  
64          assertEquals( 66, compilerError.getEndColumn() );
65  
66          assertEquals( 1, compilerError.getStartLine() );
67  
68          assertEquals( 1, compilerError.getEndLine() );
69      }
